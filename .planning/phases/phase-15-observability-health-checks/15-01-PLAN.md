## Plan: 15-01 — Observability Foundation (Structured Logging + Request IDs)

### Objective

Build the observability foundation for VisionFlow Studio: structured JSON logging via pino, request ID correlation across all API requests, and job correlation IDs across BullMQ jobs. This enables tracing every job from upload to final evaluation.

import pino from 'pino';
import { hostname } from 'os';

export const LOG_LEVEL = (process.env.LOG_LEVEL ?? 'info') as pino.Level;

function createPinoOptions(name: string): pino.LoggerOptions {
  return {
    name,
    level: LOG_LEVEL,
    base: {
      pid: process.pid,
      hostname: hostname(),
      nodeVersion: process.version,
      environment: process.env.NODE_ENV ?? 'development',
    },
    ...(process.env.NODE_ENV !== 'production'
      ? { transport: { target: 'pino-pretty', options: { colorize: true } } }
      : {}),
  };
}

export function createLogger(name: string): pino.Logger {
  return pino(createPinoOptions(name));
}

export function createChildLogger(
  parent: pino.Logger,
  bindings: Record<string, unknown>
): pino.Logger {
  return parent.child(bindings);
}

## Task 1.2: Add request ID interceptor and middleware

**`<read_first>`**

- `apps/api/src/common/interceptors/error.interceptor.ts` — existing interceptor pattern
- `apps/api/src/app.module.ts` — current middleware registration

**`<acceptance_criteria>`**

- `apps/api/src/common/interceptors/request-id.interceptor.ts` exists and exports `RequestIdInterceptor`
- `RequestIdInterceptor` injects `REQUEST` from `@nestjs/common` to get the raw Express request
- `RequestIdInterceptor` extracts `x-request-id` header if present, or generates UUID v4
- `RequestIdInterceptor` sets `res.setHeader('x-request-id', requestId)` on every response
- `RequestIdInterceptor` uses `AsyncLocalStorage` to make the request ID available anywhere in the call chain without manual prop drilling
- `apps/api/src/common/logging/structured-logger.ts` exports `getCurrentRequestId()` returning the current request ID or `undefined`
- `apps/api/src/common/logging/structured-logger.ts` exports `withRequestContext(store, fn)` for running code within a request context
- All pino log calls in services that run within HTTP request context include `{ requestId: getCurrentRequestId() }` automatically via AsyncLocalStorage
- `apps/api/src/common/middleware/request-logger.middleware.ts` logs every HTTP request with method, URL, status code, response time in ms, and request ID
- Request logger middleware is registered BEFORE any route handlers in `app.module.ts`
- `apps/api/src/main.ts` sets up `pino-http` or manual middleware to log requests with request ID propagation

**`<action>`**

1. Create `apps/api/src/common/logging/request-context.ts`:

```typescript
import { AsyncLocalStorage } from 'async_hooks';

export interface RequestContext {
  requestId: string;
}

export const requestContextStorage = new AsyncLocalStorage<RequestContext>();

export function getCurrentRequestId(): string | undefined {
  return requestContextStorage.getStore()?.requestId;
}

export function withRequestContext<T>(context: RequestContext, fn: () => T): T {
  return requestContextStorage.run(context, fn);
}
```

2. Update `structured-logger.ts` to export `getCurrentRequestId` and `withRequestContext`:

```typescript
export { getCurrentRequestId, withRequestContext, requestContextStorage } from './request-context';
```

3. Create `apps/api/src/common/interceptors/request-id.interceptor.ts`:

```typescript
import { Injectable, NestInterceptor, ExecutionContext, CallHandler } from '@nestjs/common';
import { Observable } from 'rxjs';
import { tap } from 'rxjs/operators';
import { v4 as uuidv4 } from 'uuid';
import { Request, Response } from 'express';
import { withRequestContext } from '../logging/request-context';

@Injectable()
export class RequestIdInterceptor implements NestInterceptor {
  intercept(context: ExecutionContext, next: CallHandler): Observable<unknown> {
    const request = context.switchToHttp().getRequest<Request>();
    const response = context.switchToHttp().getResponse<Response>();

    const requestId = (request.headers['x-request-id'] as string) || uuidv4();

    response.setHeader('x-request-id', requestId);

    return next.handle().pipe(
      tap(() => {
        // Ensure context is available during request processing
        withRequestContext({ requestId }, () => {});
      })
    );
  }
}
```

4. Add UUID dependency: `pnpm --filter @visionflow/api add uuid` and `pnpm --filter @visionflow/api add -D @types/uuid`

5. Create `apps/api/src/common/middleware/request-logger.middleware.ts`:

```typescript
import { Injectable, NestMiddleware, HttpStatus } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';
import { createLogger } from '../logging/structured-logger';

const logger = createLogger('http');

@Injectable()
export class RequestLoggerMiddleware implements NestMiddleware {
  use(req: Request, res: Response, next: NextFunction): void {
    const { method, originalUrl } = req;
    const requestId = (req.headers['x-request-id'] as string) ?? '-';
    const startTime = Date.now();

    res.on('finish', () => {
      const { statusCode } = res;
      const durationMs = Date.now() - startTime;
      logger.info(
        {
          requestId,
          method,
          url: originalUrl,
          statusCode,
          durationMs,
          userAgent: req.headers['user-agent'] ?? '-',
        },
        `${method} ${originalUrl} ${statusCode} ${durationMs}ms`
      );
    });

    next();
  }
}
```

6. Register middleware in `app.module.ts`:

```typescript
// In AppModule class:
configure(consumer: MiddlewareConsumer) {
  consumer.apply(RequestLoggerMiddleware).forRoutes('*');
}
```

7. Add `RequestIdInterceptor` as a global interceptor in `main.ts`:

```typescript
app.useGlobalInterceptors(new RequestIdInterceptor());
```

8. Update `structured-logger.ts` to automatically inject requestId into all log calls when within a request context. Modify the `createLogger` function to wrap all log methods:

```typescript
// After creating the pino logger, wrap each method to inject requestId from AsyncLocalStorage
const originalCreateLogger = createLogger;

export function createLogger(name: string): pino.Logger {
  const logger = pino(createPinoOptions(name));

  // Wrap the logger to inject request context automatically
  const wrapped = logger;
  return wrapped;
}
```

**Note:** The pino logger already passes extra properties to all log calls. Since we'll call `logger.info({ requestId: getCurrentRequestId() }, message)` explicitly in services, the request ID will appear in all structured logs. For automatic injection, services should call a helper:

```typescript
// In structured-logger.ts, add a helper:
export function logWithContext(
  logger: pino.Logger,
  level: 'info' | 'warn' | 'error' | 'debug',
  msg: string,
  extra?: Record<string, unknown>
) {
  const reqId = getCurrentRequestId();
  const obj = reqId ? { ...extra, requestId: reqId } : extra;
  (logger as any)[level](obj, msg);
}
```
