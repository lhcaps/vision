## Task 2.1: Design health check architecture

**`<read_first>`**

- `apps/api/src/health/health.controller.ts` — existing health controller
- `apps/api/src/health/health.module.ts` — existing health module

**`<acceptance_criteria>`**

- Health module uses NestJS `@nestjs/terminus` library for health check patterns
- Health check controller has two distinct endpoints: `/api/health/live` and `/api/health/deep`
- `/api/health/live` returns HTTP 200 immediately without checking any dependencies (just `ok: true`)
- `/api/health/deep` checks all dependencies and returns HTTP 200 only when all pass, HTTP 503 when any fail
- Each dependency check has a timeout of 5000ms (configurable via `HEALTH_CHECK_TIMEOUT_MS`)
- All health check results include: `status`, `responseTimeMs`, and dependency-specific details
- Health check response DTOs are typed and exported

3. Create individual health check services (dependency injection friendly):

**PostgresHealthService** — uses Prisma to query `SELECT 1`:

```typescript
@Injectable()
export class PostgresHealthService {
  constructor(private prisma: PrismaService) {}
  async check(timeoutMs = 5000): Promise<DependencyHealthDto> {
    const start = Date.now();
    try {
      await Promise.race([
        this.prisma.$queryRaw`SELECT 1`,
        new Promise((_, reject) => setTimeout(() => reject(new Error('Timeout')), timeoutMs)),
      ]);
      return { status: 'up', responseTimeMs: Date.now() - start };
    } catch (err) {
      return {
        status: 'down',
        responseTimeMs: Date.now() - start,
        details: { error: (err as Error).message },
      };
    }
  }
}
```

**RedisHealthService** — uses BullMQ's Redis connection:

```typescript
@Injectable()
export class RedisHealthService {
  constructor(private jobQueue: JobQueue) {}
  async check(timeoutMs = 5000): Promise<DependencyHealthDto> {
    const start = Date.now();
    try {
      // Use the job queue's Redis client if available
      const client = (this.jobQueue as BullMqJobQueue).getRedisClient?.();
      if (!client) return { status: 'up', responseTimeMs: 0, details: { note: 'memory mode' } };
      await Promise.race([
        client.ping(),
        new Promise((_, reject) => setTimeout(() => reject(new Error('Timeout')), timeoutMs)),
      ]);
      return { status: 'up', responseTimeMs: Date.now() - start };
    } catch (err) {
      return {
        status: 'down',
        responseTimeMs: Date.now() - start,
        details: { error: (err as Error).message },
      };
    }
  }
}
```

**MinioHealthService** — uses MinIO bucket list operation:

```typescript
@Injectable()
export class MinioHealthService {
  constructor(private storage: StorageRepository) {}
  async check(timeoutMs = 5000): Promise<DependencyHealthDto> {
    const start = Date.now();
    try {
      // Call listBuckets or a simple headObject
      await Promise.race([
        this.storage.listBuckets?.() ?? Promise.resolve(),
        new Promise((_, reject) => setTimeout(() => reject(new Error('Timeout')), timeoutMs)),
      ]);
      return { status: 'up', responseTimeMs: Date.now() - start };
    } catch (err) {
      return {
        status: 'down',
        responseTimeMs: Date.now() - start,
        details: { error: (err as Error).message },
      };
    }
  }
}
```

**CvWorkerHealthService** — calls the FastAPI `/health` endpoint:

```typescript
@Injectable()
export class CvWorkerHealthService {
  private readonly workerUrl = process.env.CV_WORKER_URL ?? 'http://localhost:8001';
  async check(timeoutMs = 5000): Promise<DependencyHealthDto> {
    const start = Date.now();
    try {
      const response = await fetch(`${this.workerUrl}/health`, {
        signal: AbortSignal.timeout(timeoutMs),
      });
      if (!response.ok) {
        return {
          status: 'down',
          responseTimeMs: Date.now() - start,
          details: { httpStatus: response.status },
        };
      }
      const data = (await response.json()) as {
        version?: string;
        capabilities?: Record<string, unknown>;
      };
      return {
        status: 'up',
        responseTimeMs: Date.now() - start,
        details: { version: data.version, capabilities: data.capabilities },
      };
    } catch (err) {
      return {
        status: 'down',
        responseTimeMs: Date.now() - start,
        details: { error: (err as Error).message },
      };
    }
  }
}
```

4. Update `health.controller.ts`:

```typescript
@Get('live')
@Header('Cache-Control', 'no-cache, no-store, must-revalidate')
async live(): Promise<LivenessResponseDto> {
  return {
    status: 'ok',
    timestamp: new Date().toISOString(),
    uptimeSeconds: process.uptime(),
  };
}

@Get('deep')
async deep(): Promise<HealthResponseDto> {
  const [postgres, redis, minio, cvWorker] = await Promise.all([
    this.postgresHealth.check(),
    this.redisHealth.check(),
    this.minioHealth.check(),
    this.cvWorkerHealth.check(),
  ]);

  const allUp = [postgres, redis, minio, cvWorker].every(d => d.status === 'up');

  const response: HealthResponseDto = {
    status: allUp ? 'healthy' : 'unhealthy',
    timestamp: new Date().toISOString(),
    uptimeSeconds: process.uptime(),
    version: process.env.npm_package_version ?? '0.0.0',
    dependencies: { postgres, redis, minio, cvWorker },
  };

  if (!allUp) {
    throw new ServiceUnavailableException(response);
  }
  return response;
}
```
